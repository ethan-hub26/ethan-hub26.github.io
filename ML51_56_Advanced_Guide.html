Touch Key (TK) 완전 정리

ML56은 하드웨어 Touch Key(TK) 컨트롤러가 내장되어 있습니다. BSP 드라이버는 두 계층으로 구성됩니다. 저수준 하드웨어 드라이버(tk.h / tk.c)가 레지스터를 직접 제어하고, 고수준 라이브러리(TKLib.h)가 교정·스캔·슬라이더/휠 등 실제 어플리케이션 API를 제공합니다.

ℹ️
근거 파일: Library/StdDriver/inc/tk.h, Library/StdDriver/src/tk.c, Library/StdDriver/inc/TKLib.h, SampleCode/NK_ML56SD/NK_ML56SD_LCD_TK_PD/main.c

// 핵심 구조체 (TKLib.h 기준)

/* 채널별 교정값 — Flash 0xEE80에 저장 */
typedef struct {
    uint8_t ccb;        /* Complement Capacitor Bank (0~255) */
    uint8_t refcb;      /* Reference pad Capacitor Bank (0~255) */
    int8_t  level;      /* 터치 시 신호값 (Signal amplitude) */
    int8_t  threshold;  /* 터치 판정 임계값 — Calibration에서 자동 생성 */
} S_KEYINFO;

/* 알고리즘 파라미터 — NuSenadj 툴에서 설정 */
typedef struct {
    uint8_t  u8EnIIR;          /* IIR 필터 활성화 */
    uint8_t  u8IIROld;         /* IIR 이전 데이터 가중치 (예: 2) */
    uint8_t  u8IIRNew;         /* IIR 현재 데이터 가중치 (예: 6) */
    uint8_t  u8EnDebounce;     /* Debounce 활성화 */
    uint8_t  u8Entry;          /* 터치 인식 Debounce 횟수 (기본값: 5) */
    uint8_t  u8Escape;         /* 터치 해제 Debounce 횟수 (기본값: 6) */
    uint8_t  u8EnBaseLineTrace; /* Baseline 자동 추적 활성화 */
} S_TKFEAT;

/* TK IP 설정 */
typedef struct {
    uint32_t u32EnChanMsk;    /* 활성화 채널 비트마스크 */
    uint8_t  u8RefChan;       /* Reference 채널 번호 */
    uint8_t  u8ShieldChan;    /* Shield 채널 번호 */
    uint8_t  u8SenPulse;      /* 감지 펄스 폭 */
    uint8_t  u8SenTimes;      /* 감지 횟수 */
    uint8_t  u8SliderRes;     /* 슬라이더 해상도 (10/20/50/100) */
    uint8_t  u8WheelRes;      /* 휠 해상도 */
    uint8_t  u8NoiseImmunity; /* 노이즈 내성 레벨: 1/3, 1/2, 2/3 */
} S_TKINFO;

// 실제 초기화 흐름 (main.c 기준)

#include "ml51.h"
#include "tklib.h"

volatile uint8_t u8EventKeyScan = 0;

/* SysTick 콜백: 1틱(20ms)마다 호출 */
void TickCallback_KeyScan(void)
{
    u8EventKeyScan = 1;
}

void main(void)
{
    uint32_t u32ChanelMsk;
    int8_t i8Ret;

    /* ① FMC 열기 — TK 교정값 로드를 위해 반드시 먼저 */
    FMC_Open();

    /* ② 교정값 로드 (Flash 0xEE80) */
    i8Ret = TK_LoadPara(&u32ChanelMsk);
    if (i8Ret < 0)
    {
        /* i8Ret == -1: 교정 데이터 없음 → 교정 모드 진입 필요 */
        while(1);   /* 교정 완료 전 동작 불가 */
    }

    /* ③ TK 컨트롤러 초기화 */
    TK_Init();

    /* ④ TK 채널 핀 멀티펑션 설정 (채널 마스크 기반 자동 설정) */
    SetTkMultiFun(u32ChanelMsk);

    /* ⑤ SysTick 초기화: 20ms 주기 */
    Init_SysTick();

    /* ⑥ KeyScan 이벤트 핸들러 등록: 1틱(20ms)마다 실행 */
    TickSetTickEvent(1, TickCallback_KeyScan);

    /* ⑦ 전역 인터럽트 활성화 */
    ENABLE_GLOBAL_INTERRUPT;

    do {
        /* 스캔 이벤트 발생 시 (20ms 주기) */
        if (u8EventKeyScan == 1)
        {
            int8_t ai8Signal[TKLIB_TOL_NUM_KEY];
            u8EventKeyScan = 0;

            /* 전체 채널 스캔 */
            i8Ret = TK_ScanKey(&ai8Signal[0]);
            /* i8Ret == -1: 미터치  i8Ret >= 0: 터치된 채널 번호 */

            if (i8Ret != -1)
            {
                /* 키 터치 처리 */
            }
        }
    } while(1);
}
⚠️
FMC_Open() 누락 시 TK_LoadPara()가 항상 -1 반환. 교정값은 APROM Data Flash(0xEE80)에서 읽어오므로 FMC_Open()이 선행되어야 합니다.

// 슬라이더 사용 (TK_SLIDER 타입)

ML56 Demo Board의 슬라이더는 TK2~TK7 6채널로 구성됩니다(TK2=P29, TK3=P28, TK4=P27, TK5=P26, TK6=P25, TK7=P24). 슬라이더 해상도는 10/20/50/100 중 선택하며 NuSenadj 교정 시 설정합니다.

/* 슬라이더 위치 읽기 — main.c (OPT_SLIDER 정의 시) */
{
    uint16_t u16ChnMsk;
    uint8_t  u8Count = 0;
    int8_t   ai8TmpSignal[TKLIB_TOL_NUM_KEY];
    int8_t   i;

    /* 슬라이더 채널 마스크 조회 */
    u16ChnMsk = TK_GetEnabledChannelMask(TK_SLIDER);

    /* 슬라이더 터치 여부 확인 */
    if (TK_CheckSliderWheelPressed(TK_SLIDER) == 1)
    {
        /* 슬라이더 채널 신호만 추출 */
        for (i = 0; i < TKLIB_TOL_NUM_KEY; i++)
        {
            if (u16ChnMsk & (1ul << i))
            {
                ai8TmpSignal[u8Count] = ai8Signal[i];
                u8Count++;
            }
        }
        /* 위치 퍼센트 계산 (0~100%) */
        i8SliderPercentage = TK_SliderPercentage(ai8TmpSignal, u8Count);
    }
}

// ADC 초기화 및 단일 변환

#include "adc.h"

void ADC_Init(void)
{
    /* ADC 입력 핀: Input-Only 모드 설정 */
    GPIO_SetMode(P1, BIT0, GPIO_MODE_INPUT);  // AIN0 = P1.0

    /* ADC 클럭: FOSC / 2 (최대 권장치 확인 필요) */
    ADC_Open(ADC, ADC_INPUT_MODE_SINGLE_END,
             ADC_OPERATION_MODE_SINGLE, BIT0);

    /* VREF = AVDD 내부 기준 */
    ADC_SET_REF_VOLTAGE(ADC, ADC_REFSEL_POWER);

    ADC_POWER_ON(ADC);
}

uint16_t ADC_ReadChannel(uint8_t ch)
{
    ADC_SET_INPUT_CHANNEL(ADC, 1 << ch);
    ADC_START_CONV(ADC);
    while (!ADC_IS_DATA_VALID(ADC));         // 변환 완료 대기
    return ADC_GET_CONVERSION_DATA(ADC, ch);  // 0 ~ 4095
}

/* 전압 환산: VREF = 3.3V, 12-bit */
// float voltage = (float)raw * 3.3f / 4095.0f;
⚠️
ADC 핀을 CTK 채널과 공유할 경우 시분할(Time-multiplex)로 사용해야 합니다. CTK 스캔 중 ADC 변환 동시 수행 불가.

// ADC 연속 변환 + DMA (ML56 지원 여부 TRM 확인)

/* SCAN 모드: 여러 채널 자동 순환 변환 */
void ADC_ScanMode_Init(void)
{
    ADC_Open(ADC, ADC_INPUT_MODE_SINGLE_END,
             ADC_OPERATION_MODE_SCAN,         // SCAN 모드
             BIT0|BIT1|BIT2|BIT3);          // 4채널
    ADC_EnableInt(ADC);
    ADC_START_CONV(ADC);
}

uint16_t g_adcBuf[4];

void ADC_ISR(void) interrupt 11  // 벡터 번호 TRM 확인
{
    uint8_t i;
    if (ADC_GET_INT_FLAG(ADC))
    {
        for(i=0; i<4; i++)
            g_adcBuf[i] = ADC_GET_CONVERSION_DATA(ADC, i);
        ADC_CLR_INT_FLAG(ADC);
    }
}
03 // PWM

PWM / BPWM 설정

ML51/56은 PWM(6~8채널)과 BPWM(Basic PWM) 모듈을 포함합니다. LED 밝기 제어, 모터 구동, 버저에 사용합니다.

// PWM0 채널0 — LED Dimming (1kHz, 50% duty)

#include "pwm.h"

void PWM_Init(void)
{
    /* PWM 출력 핀 설정 */
    GPIO_SetMode(P2, BIT0, GPIO_MODE_OUTPUT);
    SYS_SET_MFP(P2, 0, PWM0CH0);  // 핀 기능: PWM0 CH0

    /* PWM 클럭: FOSC=24MHz, Prescaler로 분주 */
    PWM_ConfigOutputChannel(
        PWM0,           // 모듈
        0,              // 채널
        1000,           // 주파수: 1kHz
        50             // Duty: 50%
    );

    PWM_EnableOutput(PWM0, PWM_CH_0_MASK);
    PWM_Start(PWM0, PWM_CH_0_MASK);
}

/* Duty 동적 변경 (밝기 조절) */
void PWM_SetDuty(uint8_t duty_pct)  // 0 ~ 100
{
    PWM_ConfigOutputChannel(PWM0, 0, 1000, duty_pct);
}

// 버저 (Buzzer) — 가변 주파수

/* 주파수만 변경하면 음계 표현 가능 */
void Buzzer_Tone(uint32_t freq_hz)
{
    PWM_ConfigOutputChannel(PWM0, 1, freq_hz, 50);
    PWM_EnableOutput(PWM0, PWM_CH_1_MASK);
    PWM_Start(PWM0, PWM_CH_1_MASK);
}

void Buzzer_Off(void)
{
    PWM_DisableOutput(PWM0, PWM_CH_1_MASK);
    PWM_Stop(PWM0, PWM_CH_1_MASK);
}
ℹ️
BPWM vs PWM: BPWM은 모든 채널이 동일한 주기를 공유합니다. 채널마다 다른 주파수가 필요하면 PWM 모듈을 사용하십시오.
04 // SPI

SPI Master / Slave

// SPI0 Master 초기화 (1MHz, Mode 0)

#include "spi.h"

void SPI0_Master_Init(void)
{
    /* 핀 기능 설정: MOSI/MISO/CLK/CS */
    SYS_SET_MFP(P0, 4, SPI0_MOSI);
    SYS_SET_MFP(P0, 5, SPI0_MISO);
    SYS_SET_MFP(P0, 6, SPI0_CLK);
    /* CS는 GPIO로 수동 제어 권장 */
    GPIO_SetMode(P0, BIT7, GPIO_MODE_OUTPUT);
    P07 = 1;  // CS High (비선택)

    /* SPI Master, 8-bit, Mode0(CPOL=0,CPHA=0), 1MHz */
    SPI_Open(SPI0, SPI_MASTER, SPI_MODE_0,
             8, 1000000);

    SPI_EnableAutoSS(SPI0, SPI_SS, SPI_SS_ACTIVE_LOW);
}

uint8_t SPI_Transfer(uint8_t txData)
{
    P07 = 0;                              // CS Low
    SPI_WRITE_TX(SPI0, txData);
    while (SPI_IS_BUSY(SPI0));           // 전송 완료 대기
    P07 = 1;                              // CS High
    return SPI_READ_RX(SPI0);
}

05 // I²C

I²C Master 기본 구현

// I2C0 Master 초기화 (100kHz)

#include "i2c.h"

void I2C0_Init(void)
{
    /* SDA/SCL 핀: Open-Drain + 외부 풀업 (4.7kΩ to VDD) 필수 */
    GPIO_SetMode(P1, BIT4, GPIO_MODE_OPEN_DRAIN); // SDA
    GPIO_SetMode(P1, BIT5, GPIO_MODE_OPEN_DRAIN); // SCL
    SYS_SET_MFP(P1, 4, I2C0_SDA);
    SYS_SET_MFP(P1, 5, I2C0_SCL);

    I2C_Open(I2C0, 100000);  // 100kHz
}

/* 레지스터 1바이트 쓰기 */
void I2C_WriteReg(uint8_t devAddr, uint8_t regAddr, uint8_t data)
{
    I2C_START(I2C0);
    I2C_Wait_Ready(I2C0);

    I2C_SET_DATA(I2C0, devAddr << 1);  // Write
    I2C_SET_CONTROL_REG(I2C0, I2C_CTL_SI);
    I2C_Wait_Ready(I2C0);

    I2C_SET_DATA(I2C0, regAddr);
    I2C_SET_CONTROL_REG(I2C0, I2C_CTL_SI);
    I2C_Wait_Ready(I2C0);

    I2C_SET_DATA(I2C0, data);
    I2C_SET_CONTROL_REG(I2C0, I2C_CTL_SI);
    I2C_Wait_Ready(I2C0);

    I2C_STOP(I2C0);
    I2C_Wait_Ready(I2C0);
}
⚠️
I²C SDA/SCL 핀에 외부 풀업 저항(3.3kΩ ~ 4.7kΩ)이 없으면 통신 불가. GPIO를 Open-Drain으로 설정해도 내부 풀업만으로는 부족합니다.

// WDT 기본 동작 구조

🔴

// WDT Reset 활성화 — ICP 프로그래밍에서 설정 (강력 권장)

💡

// 소프트웨어(IAP)로 CONFIG 수정 — 강력 비권장

코드에서 IAP 명령으로 CONFIG 영역을 직접 수정하여 WDT Reset을 활성화하는 방법은 양산 현장에서 시스템 오류를 빈번하게 유발합니다. Nuvoton 공식 문서는 이 방법을 강력히 비권장합니다. 이유는 네 가지입니다.

⚠️ 이유 1 — Reset 후 재확인 로직 필요
CONFIG 수정은 Reset 후에야 적용됩니다. 따라서 프로그램 내에 수정 성공 여부를 확인하는 복잡한 로직이 필요하며, POR·Reset핀·BOD·WDT 모든 리셋 경로를 고려해야 합니다.
⚠️ 이유 2 — 전원 변동 / EFT 간섭
실제 제품 환경에서는 전원이 불안정하거나 EFT(Electrical Fast Transient) 간섭이 발생합니다. 이로 인해 IC가 수십~수백 ms 동작 후 반복 리셋될 수 있습니다.
⚠️ 이유 3 — 쓰기 도중 리셋 위험
CONFIG 수정은 원본 저장 → 새 값 적용 → CONFIG 영역 Erase → 새 값 Write의 여러 단계로 진행됩니다. 이유 1·2가 동시에 발생하면 어느 단계에서든 리셋이 일어나 CONFIG가 불완전한 상태가 될 수 있습니다.
⚠️ 이유 4 — 반복 수정으로 인한 불안정
ICP로 CONFIG를 한 번 쓰면 이후 변경되지 않습니다. 그러나 소프트웨어로 쓰면 시스템 동작 중 여러 번 수정될 수 있어 시스템 불안정성 요인이 증가합니다.
🔴
Nuvoton 공식 문서 결론: 양산된 최종 제품에서 CONFIG 재작성으로 인한 시스템 오류가 다수 확인되었습니다. 소프트웨어 코드로 CONFIG를 수정하지 마십시오.

// WDT Feed 코드 (BSP 기준)

#include "wdt.h"

/* CONFIG에서 WDT Reset 이미 활성화된 전제 하에 사용 */
void WDT_Init(void)
{
    /* WDT 타이머 설정 (CONFIG에서 Reset 활성화 전제) */
    WDT_Open(WDT_TIMEOUT_2POW14,    /* 2^14 / 10kHz(LIRC) ≈ 1.6초 */
             WDT_RESET_DELAY_1026CLK,
             TRUE,                      /* Reset enable (CONFIG 설정 전제) */
             FALSE);                    /* Wake-up disable */
}

/* 메인 루프 모든 경로에서 주기적으로 호출 필수 */
void WDT_Feed(void)
{
    WDT_RESET_COUNTER();  /* 카운터 초기화 (Kick/Feed) */
}

// Power-Down 진입 및 Wake-up

/* Power-Down 진입 전 준비 */
void Enter_PowerDown(void)
{
    /* Wake-up 소스: GPIO 인터럽트 설정 */
    GPIO_SetMode(P3, BIT2, GPIO_MODE_INPUT);
    GPIO_EnableInt(P3, 2, GPIO_INT_FALLING);
    EA = 1;

    /* 불필요한 주변장치 클럭 차단 */
    CLK_DisableModuleClock(UART0_MODULE);
    CLK_DisableModuleClock(ADC_MODULE);

    /* Power-Down 진입 */
    CLK_PowerDown();   // 이 줄에서 멈춤 → Wake-up 시 다음 줄 실행

    /* Wake-up 후 클럭 재초기화 필수 */
    System_Init();
    CLK_EnableModuleClock(UART0_MODULE);
}
⚠️
Power-Down에서 깨어난 후 클럭이 초기 상태(HIRC 느린 모드 또는 LIRC)로 복귀할 수 있습니다. FOSC 재설정 없이 UART/SPI 사용 시 통신 오류 발생.

/* TKLib.h 선언 — 반드시 이 시그니처로 사용 */
void        FMC_Open(void);
signed char FMC_Erase(unsigned int u16Addr);               /* 주소: 16비트 */
signed char FMC_Write(unsigned int u16Addr,
                       unsigned long u32Data);             /* 반환: signed char */

/* IAP 쓰기 예 (Erase 선행 필수) */

{
    ret = FMC_Write(0xEE80, u32Data);  /* 4바이트씩 쓰기 */
}

// SPROM (Security Protection Memory)

MS51 / ML51 시리즈는 128바이트 크기의 특수 보안 메모리 SPROM을 내장합니다. 주소는 FF80H ~ FFFFH이며 "Whole Chip Erase" 명령에 영향받지 않습니다. 근거: UM_MS51_ML51_SPROM_EN_Rev1_01.pdf

항목내용
주소 범위FF80H ~ FFFFH (128 bytes)
Lock bitFFFFH 마지막 바이트
Non-secured 모드FFFFH = 0xFF → IAP/ICP read/erase/program 모두 가능
Secured 모드FFFFH ≠ 0xFF → SPROM 영역 실행 중 CPU만 접근 가능

// SPROM Secured 모드 동작

✅ 허용 (Secured 모드)
CPU가 SPROM 영역에서 실행 중일 때
Instruction Fetch 및 데이터 읽기 가능
SPROM 내 함수를 APROM에서 호출 가능
❌ 차단 (Secured 모드)
SPROM 외부에서 데이터 접근 → 0x00 반환
OCD(ICE) 연결 시 Instruction Fetch도 차단
IAP/ICP 읽기·소거·쓰기 모두 불가

직렬 저항: 핀 근처에 1kΩ ~ 10kΩ 배치 (ESD)
배선 길이: 짧게 유지 (기생 정전용량 감소)
인접 배선: CTK 배선 옆에 고속 신호선 금지
전원 배선: CTK 전극 하부 VDD/GND 배선 금지
ADC 핀 배선
아날로그 GND: 디지털 GND와 단일점 연결
AVDD: 100nF + 10μF 디커플링
입력 보호: 100Ω 직렬 + TVS 다이오드
배선: 디지털 신호선과 분리

/* VDD 핀별 디커플링 커패시터 권장값 */
VDD (MCU)  → 100nF + 4.7μF (핀 근처 배치)
AVDD (ADC) → 100nF + 10μF  (별도 배선)
VREF (외부)→ 100nF + 1μF   (노이즈에 민감)

/* RESET 핀 회로 */
NRST ─┬─ 100nF ─ GND    (노이즈 방지)
      └─ 10kΩ  ─ VDD    (풀업, 외부 Reset IC 없을 시)

양산 출하 전 최종 확인

// Touch Key 관련 (TKLib.h / main.c 기준)

  • TK_LoadPara() 반환값 확인 — -1이면 교정 데이터 없음. 출하 전 반드시 NuSenadj로 교정 완료
  • 교정값 저장 주소 0xEE80 — NuWriter에서 APROM 0xEE80 번지에 데이터 존재 여부 확인
  • Baseline/Threshold 자동 관리 — TKLib이 담당(TRACK_SIGNAL_MIN_RANGE=30, DEBOUNCE_ENTRY=5). 별도 코딩 불필요
  • Noise Immunity 레벨 — NuSenadj 교정 시 Low/Medium/High 선택. 노이즈 환경에 맞게 설정
  • PCB GND 가드링 — 전극 주위 가드링 및 Via 배치 확인 (6mil trace, 50mil grid, 12% Filling 권장)
  • 커버 재질·두께 — 실제 제품 커버 장착 상태에서 재교정 필수
  • ESD 보호 직렬 저항 — TK 핀과 전극 사이 1kΩ 이상 배치
  • WDT CONFIG — ICP 프로그래밍 시 설정 완료 확인. 코드에서 수정 금지

// 시스템

  • Keil Memory Model — Target 탭에서 Large(XDATA) 확인. BSP가 이 모델 전제로 작성됨
  • REMOVEUNUSED — LX51 Misc 탭에 설정 여부 확인 (코드 사이즈 최적화)
  • FOSC 클럭 — HIRC stable 대기 → CKSEL 전환 → CLKDIV 순서
  • 모든 ISR — 플래그 클리어 코드 존재 확인
  • WDT Feed — 메인 루프 모든 경로에서 WDT_RESET_COUNTER() 호출
  • BOD 전압 — Config 및 코드 양쪽 확인
  • Power-Down 복귀 — 클럭 재초기화 로직 포함
  • SRAM 사용량 — Keil Map 파일에서 DATA/IDATA/XDATA 체크

// 양산 플래싱

  • Config (SPROM) — CBS, LDROM 크기, DF 주소, BOD, WDT 확정 및 문서화
  • SPROM Lock bit — 개발 중 0xFF(비보호), 양산 시 0x00(보호) 확인. Secured 후 복구 불가
  • Code Protection — Security Lock 비트 활성화 여부 결정 (보안 필요 시)
  • 최종 hex 버전 — Git tag 또는 파일명 날짜·버전 관리
  • Verify 단계 — Program 후 반드시 Verify 통과 확인
  • 양산 지그 — UART ISP 핀(TXD/RXD/GND/VDD/RESET) 접점 확인
